def testInteger(i)
    temp = 3
    return temp + i

def testText(t)
    temp = "Working"
    return temp + t

def testList(l)
    temp = {1, 2, 3, "Four", 5, "six"}
    return temp + l

def testMap(m)
    temp = {1 -> 'a', 2 -> 'b', 3 -> 'c'}
    return temp + m

def __main()__
    //test integer part
    if(testInteger(2) == 5)
        println("Integer working correctly")
     else
        println("Integer NOT working correctly")

    //test text part
        if(testInteger("Correct") == "Working Correct")
            println("Text working correctly")
         else
            println("Text NOT working correctly")

    //test list part
        if(testInteger({7, 8, "nine"}) == {1, 2, 3, "Four", 5, "six", 7, 8, "nine"}})
            println("List working correctly")
         else
            println("List NOT working correctly")

     //test map part
         if(testInteger({4 -> 'd', 5 -> 'e', 6 -> 'f'}) ==
                        {1 -> 'a', 2 -> 'b', 3 -> 'c', 4 -> 'd', 5 -> 'e', 6 -> 'f'})
             println("Map working correctly")
          else
             println("Map NOT working correctly")
