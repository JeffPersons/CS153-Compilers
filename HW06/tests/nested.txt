def numStatus(Number x)
{
    Text status = ''
    if x < 100
    {
        status = 'x in range(-inf, 100)'
        if 75 < x
        {
            status = 'x in range(75, 100)'
        }
        elif 50 < x
        {
            status = 'x in range(50, 75)'
        }
        else
        {
            status = 'x in range(-inf, 50]'
        }
    }
    else
    {
        return 'I\'m a number 100 or bigger!!!'
    }
    return x
}
